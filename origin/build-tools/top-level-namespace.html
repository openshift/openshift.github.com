def get_branch
  branch_str = `git status | head -n1`.chomp
  branch_str =~ /.*branch (.*)/
  branch = $1 ? $1 : 'origin/master'
  return branch
end

def guess_os(base_os=nil)
  return base_os unless base_os.nil?
  if File.exist?("/etc/fedora-release")
    return "fedora"
  elsif File.exist?("/etc/redhat-release")
    data = File.read("/etc/redhat-release")
    if data.match(/centos/)
      return "centos"
    else
      return "rhel"
    end
  end
end

def send_verified_email(image_id, image_name)
  msg = "From: Jenkins <noreply@redhat.com>\nTo: Libra Team <libra-devel@redhat.com>\nSubject: [Jenkins] DevEnv Image \#{image_name} (\#{image_id}) is QE Ready\n\nImage \#{image_name} (\#{image_id}) has passed validation tests and is ready for QE.\n\n"   

  Net::SMTP.start('localhost') do |smtp|
    smtp.send_message msg, "noreply@redhat.com", "libra-devel@redhat.com"
  end
end
