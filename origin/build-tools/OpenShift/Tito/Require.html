def name
  @name ||= value.
    gsub(/\(/, '-').  # package names get unparethesized
    gsub(/\)/, '').
    gsub(/>=.+/, ''). # strip version qualifiers
    gsub(/=.+/,'').
    gsub(/,/, '').
    gsub('%{?scl:%scl_prefix}', 'ruby193-').strip
end
