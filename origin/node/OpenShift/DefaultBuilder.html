def post_receive(options)
  @container.cartridge_model.do_control('pre-repo-archive',
                                        @container.cartridge_model.primary_cartridge,
                                        out:                       options[:out],
                                        err:                       options[:err],
                                        pre_action_hooks_enabled:  false,
                                        post_action_hooks_enabled: false)

  ApplicationRepository.new(@container.user).archive

  @container.build(out: options[:out],
                   err: options[:err])

  @container.start_gear(secondary_only: true,
                        user_initiated: true,
                        hot_deploy:     options[:hot_deploy],
                        out:            options[:out],
                        err:            options[:err])

  @container.deploy(out: options[:out],
                    err: options[:err])

  @container.start_gear(primary_only:   true,
                        user_initiated: true,
                        hot_deploy:     options[:hot_deploy],
                        out:            options[:out],
                        err:            options[:err])

  @container.post_deploy(out: options[:out],
                         err: options[:err])
end
